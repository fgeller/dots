   Collection of sources that we want fast access to.

     (defun helm-jump ()
       (interactive)
         (helm :sources `(
                          helm-source-buffers-list
			  ,(helm-make-source "Git files" 'helm-ls-git-source)
                          helm-source-recentf
                          helm-source-files-in-current-dir
                          helm-source-buffer-not-found
                          )))
   #+end_src
