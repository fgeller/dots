  #+begin_src emacs-lisp
    (use-package elisp-slime-nav
      :ensure elisp-slime-nav
      :commands elisp-slime-nav-mode)
  #+end_src

  My customizations for Emacs LISP buffers.

  #+begin_src emacs-lisp
    (defun emacs-lisp-customizations ()

      (elisp-slime-nav-mode 1))

  #+begin_src emacs-lisp
